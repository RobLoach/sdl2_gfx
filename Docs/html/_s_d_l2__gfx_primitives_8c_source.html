01037                 */

01053                                 * Accumulator turned over, advance y 
01054                                 */

01056                                 y0p1++;
01057                         }
01058                         xx0 += xdir;    /* x-major so always advance X */
01059                         /*
01060                         * the AAbits most significant bits of erracc give us the intensity
01061                         * weighting for this pixel, and the complement of the weighting for
01062                         * the paired pixel. 
01063                         */

01066                         result |= pixelRGBAWeight (renderer, xx0, y0p1, r, g, b, a, wgt);

01077                 */
01078                 result |= pixelRGBA (renderer, x2, y2, r, g, b, a);
01079         }
01080 
01081         return (result);

01334                                         drawoct &= 255 - (1 << oct);

01338                         else                      drawoct |= (1 << oct);

01341                 }

01347         * Set color 
01348         */

01350         result |= SDL_SetRenderDrawBlendMode(renderer, (a == 255) ? SDL_BLENDMODE_NONE : SDL_BLENDMODE_BLEND);
01351         result |= SDL_SetRenderDrawColor(renderer, r, g, b, a);

01359                 if (cx > 0) {
01360                         xpcx = x + cx;
01361                         xmcx = x - cx;
01362

01364                         if (drawoct & 4)  result |= pixel(renderer, xmcx, ypcy);
01365                         if (drawoct & 2)  result |= pixel(renderer, xpcx, ypcy);
01366                         if (drawoct & 32) result |= pixel(renderer, xmcx, ymcy);
01367                         if (drawoct & 64) result |= pixel(renderer, xpcx, ymcy);
01368                 } else {
01369                         if (drawoct & 96) result |= pixel(renderer, x, ymcy);
01370                         if (drawoct & 6)  result |= pixel(renderer, x, ypcy);
01371                 }
01372

01376                         ypcx = y + cx;
01377                         ymcx = y - cx;

01380                         if (drawoct & 16)  result |= pixel(renderer, xmcy, ymcx);

01394                         else                                               drawoct |= (1 << startoct);

01403                 */

01406                         d_e += 2;
01407                         d_se += 2;

01410                         d_e += 2;
01411                         d_se += 4;
01412                         cy--;
01413                 }

01415         } while (cx <= cy);
01416

01530 int _ellipseRGBA(SDL_Renderer * renderer, Sint16 x, Sint16 y, Sint16 rx, Sint16 ry, Uint8 r, Uint8 g, Uint8 b, Uint8 a, Sint32 f)
01531 {

01534     Sint32 error;
01535     Sint32 curX, curY, curXp1, curYm1;
01536         Sint32 scrX, scrY, oldX, oldY;
01537     Sint32 deltaX, deltaY;
01538 
01539         /*
01540         * Sanity check radii 
01541         */
01542         if ((rx < 0) || (ry < 0)) {
01543                 return (-1);
01544         }
01545 
01546         /*
01547         * Set color
01548         */
01549         result = 0;
01550         result |= SDL_SetRenderDrawBlendMode(renderer, (a == 255) ? SDL_BLENDMODE_NONE : SDL_BLENDMODE_BLEND);
01551         result |= SDL_SetRenderDrawColor(renderer, r, g, b, a);
01552 
01553         /*
01554         * Special cases for rx=0 and/or ry=0: draw a hline/vline/pixel 
01555         */
01556         if (rx == 0) {
01557                 if (ry == 0) {
01558                         return (pixel(renderer, x, y));
01559                 } else {
01560                         return (vline(renderer, x, y - ry, y + ry));
01561                 }
01562         } else {
01563                 if (ry == 0) {
01564                         return (hline(renderer, x - rx, x + rx, y));
01565                 }
01566         }
01567 
01568         /*
01569          * Top/bottom center points.
01570          */
01571         oldX = scrX = 0;
01572         oldY = scrY = ry;
01573         result |= _drawQuadrants(renderer, x, y, 0, ry, f);
01574

01576         rx *= ELLIPSE_OVERSCAN;
01577         ry *= ELLIPSE_OVERSCAN;

01579         rx22 = rx2 + rx2;

01581         ry22 = ry2 + ry2;

01583     curY = ry;
01584     deltaX = 0;
01585     deltaY = rx22 * curY;
01586  
01587         /* Points in segment 1 */ 
01588     error = ry2 - rx2 * ry + rx2 / 4;
01589     while (deltaX <= deltaY)
01590     {
01591           curX++;
01592           deltaX += ry22;
01593  
01594           error +=  deltaX + ry2; 
01595           if (error >= 0)
01596           {
01597                curY--;
01598                deltaY -= rx22; 
01599                error -= deltaY;
01600           }
01601 
01602                   scrX = curX/ELLIPSE_OVERSCAN;
01603                   scrY = curY/ELLIPSE_OVERSCAN;
01604                   if ((scrX != oldX && scrY == oldY) || (scrX != oldX && scrY != oldY)) {
01605                         result |= _drawQuadrants(renderer, x, y, scrX, scrY, f);
01606                         oldX = scrX;
01607                         oldY = scrY;
01608                   }
01609     }
01610 
01611         /* Points in segment 2 */
01612         if (curY > 0) 
01613         {
01614                 curXp1 = curX + 1;
01615                 curYm1 = curY - 1;
01616                 error = ry2 * curX * curXp1 + ((ry2 + 3) / 4) + rx2 * curYm1 * curYm1 - rx2 * ry2;
01617                 while (curY > 0)
01618                 {
01619                         curY--;
01620                         deltaY -= rx22;
01621 
01622                         error += rx2;
01623                         error -= deltaY;
01624  
01625                         if (error <= 0) 
01626                         {
01627                curX++;
01628                deltaX += ry22;
01629                error += deltaX;
01630                         }
01631 
01632                     scrX = curX/ELLIPSE_OVERSCAN;
01633                     scrY = curY/ELLIPSE_OVERSCAN;
01634                     if ((scrX != oldX && scrY == oldY) || (scrX != oldX && scrY != oldY)) {
01635                                 oldY--;
01636                                 for (;oldY >= scrY; oldY--) {
01637                                         result |= _drawQuadrants(renderer, x, y, scrX, oldY, f);
01638                                         /* prevent overdraw */
01639                                         if (f) {
01640                                                 oldY = scrY - 1;
01641                                         }
01642                                 }
01643                                 oldX = scrX;
01644                                 oldY = scrY;
01645                     }           
01646                 }
01647 
01648                 /* Remaining points in vertical */
01649                 if (!f) {
01650                         oldY--;
01651                         for (;oldY >= 0; oldY--) {
01652                                 result |= _drawQuadrants(renderer, x, y, scrX, oldY, f);
01653                         }
01654                 }
01655         }
01656 
01657         return (result);
01658 }
